<Note>
  This endpoint uses a standard **project API key** (`AS_` prefix). Messages are scoped to the project the key belongs to. Blocked and unrouted messages are excluded unless `includeBlocked=true` is set.
</Note>

```java Java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URL;

public class ListMessages {
    public static void main(String[] args) {
        try {
            URL url = new URL("https://api.autosend.com/v1/inbound/messages?page=1&limit=50");
            HttpURLConnection con = (HttpURLConnection) url.openConnection();

            con.setRequestMethod("GET");
            con.setRequestProperty("Authorization", "Bearer as_your-api-key");

            BufferedReader in = new BufferedReader(new InputStreamReader(con.getInputStream()));
            String inputLine;
            StringBuilder content = new StringBuilder();
            while ((inputLine = in.readLine()) != null) {
                content.append(inputLine);
            }
            in.close();
            con.disconnect();

            System.out.println(content.toString());
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
```

#### Authorizations
<ParamField path="Authorizations" type="string | header" required>
  Project API key header of the form Bearer `as_<key>`. You can also pass the key via the `x-api-key` header.
</ParamField>

### Query Parameters

<ParamField query="domainId" type="string">
  Filter to messages received on a specific domain (MongoDB ObjectId of the project's `sesDomains` entry).

  Example: `"60d5ec49f1b2c72d9c8b5678"`
</ParamField>

<ParamField query="from" type="string">
  Filter by sender email address (exact match, case-insensitive).

  Example: `"customer@gmail.com"`
</ParamField>

<ParamField query="to" type="string">
  Filter by recipient email address (exact match, case-insensitive).

  Example: `"help@support.example.com"`
</ParamField>

<ParamField query="threadId" type="string">
  Filter to messages belonging to a specific conversation thread (MongoDB ObjectId).

  Example: `"60d5ec49f1b2c72d9c8b1234"`
</ParamField>

<ParamField query="search" type="string">
  Case-insensitive substring search against the message subject.

  Maximum length: `200`
</ParamField>

<ParamField query="dateFrom" type="string">
  Return only messages created on or after this timestamp (ISO 8601).

  Example: `"2026-06-01T00:00:00.000Z"`
</ParamField>

<ParamField query="dateTo" type="string">
  Return only messages created on or before this timestamp (ISO 8601).

  Example: `"2026-06-30T23:59:59.000Z"`
</ParamField>

<ParamField query="includeBlocked" type="boolean">
  When `true`, also include `BLOCKED_UNVERIFIED` and `BLOCKED_UNROUTED` messages. Defaults to `false`.

  Example: `false`
</ParamField>

<ParamField query="page" type="integer">
  Page number (1-based). Defaults to `1`.

  Minimum: `1`
</ParamField>

<ParamField query="limit" type="integer">
  Number of messages per page. Defaults to `50`.

  Minimum: `1`

  Maximum: `200`
</ParamField>
