<Note>
  This endpoint accepts a **project API key** (`AS_` prefix). Returns the delivery attempt history for a webhook, most recent first.
</Note>

<RequestExample>

```bash cURL
curl --request GET \
  --url 'https://api.autosend.com/v1/webhooks/60d5ec49f1b2c72d9c8b1234/logs?page=1&limit=50' \
  --header 'Authorization: Bearer AS_your-api-key'
```

```java Java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URL;

public class ListDeliveryLogs {
    public static void main(String[] args) {
        try {
            String webhookId = "60d5ec49f1b2c72d9c8b1234";
            URL url = new URL("https://api.autosend.com/v1/webhooks/" + webhookId + "/logs?page=1&limit=50");
            HttpURLConnection con = (HttpURLConnection) url.openConnection();

            con.setRequestMethod("GET");
            con.setRequestProperty("Authorization", "Bearer AS_your-api-key");

            BufferedReader in = new BufferedReader(new InputStreamReader(con.getInputStream()));
            String inputLine;
            StringBuilder content = new StringBuilder();
            while ((inputLine = in.readLine()) != null) {
                content.append(inputLine);
            }
            in.close();
            System.out.println(content.toString());

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
```

#### Authorizations
<ParamField path="Authorizations" type="string | header" required>
  Project API key header of the form Bearer `AS_<key>`.
</ParamField>

### Path Parameters

<ParamField path="id" type="string" required>
  The unique identifier of the webhook.

  Example: `"60d5ec49f1b2c72d9c8b1234"`
</ParamField>

### Query Parameters

<ParamField query="page" type="integer">
  Page number for pagination.

  Default: `1`
</ParamField>

<ParamField query="limit" type="integer">
  Number of logs to return per page.

  Default: `50`

  Range: `1`–`100`
</ParamField>

#### Response

<span className='text-sm'>Delivery logs retrieved successfully (200)</span>

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object">
  Paginated delivery logs

  <Expandable title="child attributes">
  <ResponseField name="data.logs" type="object[]">
    Array of delivery attempt records

    <Expandable title="log attributes">
    <ResponseField name="event" type="string">
      The event type that was delivered
    </ResponseField>

    <ResponseField name="url" type="string">
      The destination URL the payload was POSTed to
    </ResponseField>

    <ResponseField name="statusCode" type="integer | null">
      HTTP status code returned by the destination
    </ResponseField>

    <ResponseField name="success" type="boolean">
      Whether the delivery was considered successful
    </ResponseField>

    <ResponseField name="attempts" type="integer">
      Number of delivery attempts made
    </ResponseField>

    <ResponseField name="duration" type="integer | null">
      Time taken to deliver, in milliseconds
    </ResponseField>

    <ResponseField name="errorMessage" type="string | null">
      Error detail if the delivery failed
    </ResponseField>

    <ResponseField name="deliveredAt" type="string | null">
      ISO 8601 timestamp of successful delivery
    </ResponseField>
    </Expandable>
  </ResponseField>

  <ResponseField name="data.total" type="integer">
    Total number of log entries
  </ResponseField>

  <ResponseField name="data.totalPages" type="integer">
    Total number of pages
  </ResponseField>
  </Expandable>
</ResponseField>
